// Helper function to convert bit values to readable format
private string convert_to_bits(string input)
{
    string output = "";
    BitArray fud_bits = new BitArray(new byte[] { Convert.ToByte(input) });
             
    for (int x = 0; x < 8; x++) 
    {
        if (fud_bits[x] == true)
        {
            if (!output.Equals(""))
                output += "_";

            output += x.ToString(); 
        }
    }

    if (output.Equals(""))
        output = "NONE";
    else
        output += "_ON";
                      
    return output;
}
