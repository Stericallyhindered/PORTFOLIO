        private int read_single_byte(string send)

        {
            int temp = 0;
            int counter = 0;

            try
            {

                //Clear out input buffer
                if (bluetooth_port.Equals(true))
                {
                    byte_counter = 0;
                    byte_counter_end = 0;
                }
                else
                {
                    Port1.ReadExisting();
                }

                if (!send.Equals(""))
                    SERout(send);

                //Wait a second for reply
                while (counter++ < 75)
                {
                    if (bluetooth_port.Equals(true) && byte_counter_end > 0)
                    {
                        break;
                    }
                    else
                    {
                        if (Port1.BytesToRead > 0)
                            break;
                    }

                    Thread.Sleep(20);
                }


                if (bluetooth_port.Equals(true))
                    temp = byteIN[0];
                else
                    temp = Port1.ReadChar();


                if (temp.Equals(""))
                {
                    return 88;
                }
                else
                {
                    return temp;
                }
            }
            catch
            {
                return 88; //failed
            }
                
 
        }

        private string read_string(string send)
        {
          
            int counter = 0;

            //Clear existing data    

            if (!send.Equals(""))
            {
                //Clear out input buffer
                if (bluetooth_port.Equals(true))
                {
                    byte_counter = 0;
                    byte_counter_end = 0;
                }
                else
                {
                    Port1.ReadExisting();
                }

                SERout(send);
            }

            Thread.Sleep(500);

            //Wait a second for reply
            while (counter++ < 200)
            {
                if (bluetooth_port.Equals(true) && byte_counter_end > 0)
                {
                    break;
                }
                else
                {
                    if (Port1.BytesToRead > 0)
                        break;
                }

                Thread.Sleep(25);
            }

            if (counter.Equals(50))
            {
                return "";
            }
            else
            {
                if (bluetooth_port.Equals(true))
                {
                    byte[] temp_byte = new byte[byte_counter_end];
                  
                    Buffer.BlockCopy(byteIN,0,temp_byte,0,byte_counter_end);
                    return System.Text.Encoding.ASCII.GetString(temp_byte);

                }
                else
                {
                    return Port1.ReadExisting();
                }
            }
