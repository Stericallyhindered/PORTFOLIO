  <script>
    /********************************************************
     * GLOBALS
     ********************************************************/
    let allCsvMatrix  = [];
    let headers       = [];
    let dataRows      = [];
    let visibleSeries = [];
    let seriesColors  = [];
    let chartInstance = null;
    let tooltipLocked = false;
    const fileInput        = document.getElementById('file-input');
    const noFileMessage    = document.getElementById('no-file-message');
    const appUI            = document.getElementById('app-ui');
    const uploadedFileInfo = document.getElementById('uploaded-file-info');
    const controlsList     = document.getElementById('controls-list');
    const chartCanvas      = document.getElementById('myChart');

    /********************************************************
     * EVENT LISTENERS
     ********************************************************/
    fileInput.addEventListener('change', handleFileUpload);

    /********************************************************
     * FILE UPLOAD & CSV PARSING
     ********************************************************/
    async function handleFileUpload(e) {
      const file = e.target.files[0];
      if (!file) return;

      const rawCsv = await file.text();
      const rows = rawCsv.split(/\r?\n/).filter(r => r.trim() !== "");
      allCsvMatrix = rows.map(r => r.split(","));

      // We assume row 0 is headers, rest is data
      headers = allCsvMatrix[0];
      const rawData = allCsvMatrix.slice(1);
      dataRows = rawData.map(line => {
        const rowObj = {};
        headers.forEach((h, i) => {
          const val = parseFloat(line[i]);
          rowObj[h] = isNaN(val) ? (line[i] || "") : val;
        });
        return rowObj;
      });

      // Default: all columns except first one is data
      // We'll assume first column is X-axis
      visibleSeries = headers.slice(1).map(() => true);

      // Assign random color to each
      seriesColors = headers.slice(1).map(() => {
        return '#' + Math.floor(Math.random() * 16777215).toString(16).padStart(6, '0');
      });

      uploadedFileInfo.textContent = "Uploaded File: " + file.name;

      noFileMessage.style.display = "none";
      appUI.style.display         = "block";

      buildControls();
      buildChart();
    }

    /********************************************************
     * BUILD CONTROLS + CHART
     ********************************************************/
    function buildControls() {
      controlsList.innerHTML = "";
      headers.slice(1).forEach((colName, idx) => {
        const rowDiv = document.createElement("div");
        rowDiv.className = "control-row";

        const colorInput = document.createElement("input");
        colorInput.type  = "color";
        colorInput.value = seriesColors[idx];
        colorInput.addEventListener("input", () => {
          seriesColors[idx] = colorInput.value;
          buildChart();
        });
        rowDiv.appendChild(colorInput);

        const labelSpan = document.createElement("span");
        labelSpan.textContent = colName;
        rowDiv.appendChild(labelSpan);

        const toggleCheckbox = document.createElement("input");
        toggleCheckbox.type   = "checkbox";
        toggleCheckbox.checked = visibleSeries[idx];
        toggleCheckbox.addEventListener("change", () => {
          visibleSeries[idx] = toggleCheckbox.checked;
          buildChart();
        });
        rowDiv.appendChild(toggleCheckbox);

        controlsList.appendChild(rowDiv);
      });
    }

    function buildChart() {
      if (chartInstance) chartInstance.destroy();

      const xHeader = headers[0];
      const datasets = headers.slice(1).map((colName, idx) => {
        if (!visibleSeries[idx]) return null;
        return {
          label: colName,
          data: dataRows.map(r => ({ x: r[xHeader], y: r[colName] })),
          borderColor: seriesColors[idx],
          backgroundColor: seriesColors[idx],
          fill: false,
          tension: 0.2,
          borderWidth: 2,
          pointRadius: 1.5,
          pointHitRadius: 10,
          pointHoverRadius: 6
        };
      }).filter(Boolean);

      chartInstance = new Chart(chartCanvas, {
        type: "line",
        data: { datasets },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          interaction: { mode: "x", intersect: false },
          scales: {
            y: {
              type: "linear",
              display: true,
              title: { display: true, text: "Main Values", color: "#ffffff" },
              grid: { color: "#ffffff33" },
              ticks: { color: "#ffffff" }
            },
            x: {
              type: "linear",
              display: true,
              title: { display: true, text: xHeader, color: "#ffffff" },
              grid: { color: "#ffffff33" },
              ticks: { color: "#ffffff" }
            }
          },
          plugins: {
            tooltip: { enabled: true, mode: "x", intersect: false },
            legend: { display: false }
          }
        }
      });

      // Optional: "sticky" tooltip on click
      chartCanvas.onclick = evt => {
        const points = chartInstance.getElementsAtEventForMode(evt, "index", { intersect: false }, false);
        if (points.length) {
          if (tooltipLocked) {
            chartInstance.tooltip.setActiveElements([], { x: 0, y: 0 });
            tooltipLocked = false;
          } else {
            chartInstance.tooltip.setActiveElements(
              points.map(pt => ({ datasetIndex: pt.datasetIndex, index: pt.index })),
              { x: points[0].element.x, y: points[0].element.y }
            );
            tooltipLocked = true;
          }
          chartInstance.update();
        }
      };
    }
